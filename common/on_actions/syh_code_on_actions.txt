on_game_started = {
	on_actions = {
		socialism_for_the_world
		create_soviet_union
		create_german_soviet_republic

		create_soviet_bloc
	}
}

socialism_for_the_world = {
	effect = {
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
			}
		}
		give_socialism = yes
	}
}

create_soviet_union = {
	effect = {
		c:RUS ?= { set_up_soviet_gov = yes }
	}
}

create_german_soviet_republic = {
	effect = {
		c:PRU ?= { change_tag = GER }
		c:GER ?= { set_up_soviet_gov = yes }
	}
}

create_soviet_bloc = {
	effect = {
		c:RUS ?= {
			create_diplomatic_pact = {
				country = c:GER
				type = constituent_republic
			}
		}
	}
}
