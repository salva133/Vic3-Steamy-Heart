on_game_started = {
	on_actions = {
		socialism_for_the_world
		create_soviet_union
		create_soviet_bloc
		reorganise_states
		create_contrabolshevist_italy
		create_yugoslavian_federation
		create_romanian_soviet_republic
		create_white_russian_soviet_republic
		create_polish_soviet_republic
		create_ukrainian_soviet_republic
	}
}

socialism_for_the_world = {
	effect = {
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
			}
		}
		give_socialism = yes
	}
}

create_soviet_union = {
	effect = {
		c:RUS ?= {
			set_up_soviet_gov = yes
			create_power_bloc = {
				name = eastern_bloc
				map_color = { 200 0 0 }
				founding_date = 1800.1.1
				identity = identity_ideological_union
				principle = principle_creative_legislature_1
			}
		}
	}
}

reorganise_states = {
	effect = {
		every_state = {
			limit = {
				owner = {
					country_has_primary_culture = cu:north_german
				}
			}
			set_state_owner = c:GER
		}
		every_state = {
			limit = {
				owner = {
					country_has_primary_culture = cu:south_german
				}
			}
			set_state_owner = c:AUS
		}
	}
}

create_contrabolshevist_italy = {
	effect = {
		c:PAP ?= {
			change_tag = ITA
		}
		every_state = {
			limit = {
				is_homeland_of_country_cultures = c:ITA
			}
			set_state_owner = c:ITA
		}
		c:ITA ?= {
			set_up_fascist_gov = yes
		}
	}
}

create_yugoslavian_federation = {
	effect = {
		c:SER ?= {
			make_independent = yes
			change_tag = YUG
		}
		every_state = {
			limit = {
				is_homeland_of_country_cultures = c:YUG
			}
			set_state_owner = c:YUG
		}
		c:YUG ?= {
			set_up_soft_commie_gov = yes
		}
		s:STATE_BULGARIA = {
			every_scope_state = {
				save_scope_as = capital_state
			}
		}
		create_country = {
			tag = BUL
			origin = c:YUG
			state = scope:capital_state
			on_created = {
				set_up_soft_commie_gov = yes
			}
		}
		every_state = {
			limit = {
				is_homeland_of_country_cultures = c:BUL
			}
			set_state_owner = c:BUL
		}
		c:YUG = {
			create_diplomatic_pact = {
				country = c:BUL
				type = constituent_republic
			}
		}
	}
}
