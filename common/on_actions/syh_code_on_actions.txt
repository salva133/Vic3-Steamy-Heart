on_game_started = {
	on_actions = {
		socialism_for_the_world
		create_soviet_union
		create_soviet_republics
		create_soviet_bloc
		reorganise_states
		create_contrabolshevist_italy
	}
}

socialism_for_the_world = {
	effect = {
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
			}
		}
		give_socialism = yes
	}
}

create_soviet_union = {
	effect = {
		c:RUS ?= {
			set_up_soviet_gov = yes
			create_power_bloc = {
				name = eastern_bloc
				map_color = { 200 0 0 }
				founding_date = 1800.1.1
				identity = identity_ideological_union
				principle = principle_creative_legislature_1
			}
		}
	}
}

create_soviet_republics = {
	effect = {
		c:PRU ?= { change_tag = GER }
		c:GER ?= { set_up_soviet_gov = yes }
		c:AUS ?= { set_up_soviet_gov = yes }
		c:HUN ?= { set_up_soviet_gov = yes }
		c:UBD ?= { set_up_soviet_gov = yes }
		c:FIN ?= { set_up_soviet_gov = yes }
		c:KZH ?= { set_up_soviet_gov = yes }
		c:OZH ?= { set_up_soviet_gov = yes }
		c:UZH ?= { set_up_soviet_gov = yes }
		c:KOK ?= { set_up_soviet_gov = yes }
		c:BUK ?= { set_up_soviet_gov = yes }
		c:KHI ?= { set_up_soviet_gov = yes }
		c:CIR ?= { set_up_soviet_gov = yes }
		c:CHC ?= { set_up_soviet_gov = yes }
	}
}

reorganise_states = {
	effect = {
		every_state = {
			limit = {
				owner = {
					country_has_primary_culture = cu:north_german
				}
			}
			set_state_owner = c:GER
		}
		every_state = {
			limit = {
				owner = {
					country_has_primary_culture = cu:south_german
				}
			}
			set_state_owner = c:AUS
		}
	}
}

create_contrabolshevist_italy = {
	effect = {
		c:PAP ?= {
			change_tag = ITA
		}
		every_state = {
			limit = {
				is_homeland_of_country_cultures = c:ITA
			}
			set_state_owner = c:ITA
		}
		c:ITA ?= {
			set_up_fascist_gov = yes
		}
	}
}
