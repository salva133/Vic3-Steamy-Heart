give_all_states_strong_communist_bias = {
	every_scope_state = {
		limit = {
			NOT = { has_modifier = modifier_strong_communist_bias }
		}
		add_modifier = {
			name = modifier_strong_communist_bias
			years = 100
			is_decaying = yes
		}
	}
}

make_all_politicians_communist = {
	every_scope_politician = {
		limit = {
			NOT = { has_ideology = ideology:ideology_communist }
		}
		set_ideology = ideology:ideology_communist
	}
}
