	c:ITA ?= {
		create_building_and_government_buildings = yes
		create_character = {
			is_general = yes
		}
		create_character = {
			is_general = yes
		}
		create_character = {
			is_admiral = yes
		}
	}
}

create_building_and_government_buildings = {
	every_scope_state = {
		create_building = {
			building = building_construction_sector
			level = 10
		}
		if = {
			limit = {
				is_capital = no
			}
			create_building = {
				building = building_government_administration
				level = 2
			}
		}
	}
	capital = {
		create_building = {
			building = building_government_administration
			level = 20
		}
	}
}

give_all_states_strong_communist_bias = {
	every_scope_state = {
		limit = {
			NOT = { has_modifier = modifier_strong_communist_bias }
		}
		add_modifier = {
			name = modifier_strong_communist_bias
			years = 100
			is_decaying = yes
		}
	}
}

make_all_politicians_communist = {
	every_scope_politician = {
		limit = {
			NOT = { has_ideology = ideology:ideology_communist }
		}
		set_ideology = ideology:ideology_communist
	}
}

TRY_INJECT:remove_subject_diplomatic_pact = {
	if = {
		limit = {
			this = {
				has_diplomatic_pact = {
					who = root
					type = constituent_republic
				}
			}
		}
		this = {
			remove_diplomatic_pact = {
				country = root
				type = constituent_republic
			}
		}
	}
}

		}
	}
}
