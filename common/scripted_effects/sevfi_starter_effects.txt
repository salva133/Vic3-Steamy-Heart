era_3_for_the_world = {
	every_country = {
		limit = {
			NOT = { is_country_type = decentralized }
		}
		add_era_researched = era_3
		make_pops_literate_40_percent = yes # Pops need to be able to comprehend this
	}
}

create_soviet_bloc = {
	c:AUS ?= { set_up_hard_commie_gov = yes }
	c:HUN ?= { set_up_hard_commie_gov = yes }
	if = {
		limit = { has_game_rule = create_ussr }
		c:RUS ?= { set_up_hard_commie_gov = yes }
		c:FIN ?= { make_independent = yes }
		# Bind all constituent republics to c:RUS. make_ussr_constituent
		# also runs set_up_hard_commie_gov on each subject, so the AUS/HUN
		# conversion above becomes an idempotent no-op in this branch.
		c:NGF ?= { make_ussr_constituent = yes }
		c:SGF ?= { make_ussr_constituent = yes }
		c:AUS ?= { make_ussr_constituent = yes }
		c:HUN ?= { make_ussr_constituent = yes }
		c:UBD ?= { make_ussr_constituent = yes }
		c:KZH ?= { make_ussr_constituent = yes }
		c:OZH ?= { make_ussr_constituent = yes }
		c:UZH ?= { make_ussr_constituent = yes }
		c:KOK ?= { make_ussr_constituent = yes }
		c:BUK ?= { make_ussr_constituent = yes }
		c:KHI ?= { make_ussr_constituent = yes }
		c:CIR ?= { make_ussr_constituent = yes }
		c:CHC ?= { make_ussr_constituent = yes }
	}
}

create_pb_esu = {
	if = {
		limit = { has_game_rule = create_esu }
		c:NGF ?= {
			create_power_bloc = {
				name = pb_european_socialist_union
				map_color = { 89 39 32 }
				founding_date = 1836.1.1
				identity = identity_ideological_union
				principle = principle_ideological_truth_3
				principle = principle_market_unification_3
				principle = principle_transport_3
				principle = principle_freedom_of_movement_3
			}
			power_bloc ?= {
				add_cohesion_percent = 100
			}
		}
		# Bind each constituent of the ESU to NGF, unless it is already
		# a subject of NGF (left as-is to preserve any pre-existing pact).
		every_country = {
			limit = {
				OR = {
					this = c:AUS  this = c:HUN  this = c:SGF
					this = c:YUG  this = c:POL  this = c:BOH
					this = c:SLV  this = c:LIT  this = c:ROM
				}
				NOT = { is_subject_of = c:NGF }
			}
			save_scope_as = new_constituent
			c:NGF ?= { bind_constituent_republic = yes }
		}
		# ALB binds to YUG, not NGF.
		c:ALB ?= {
			if = {
				limit = { NOT = { is_subject_of = c:YUG } }
				save_scope_as = new_constituent
				c:YUG ?= { bind_constituent_republic = yes }
			}
		}
	}
}

create_socialist_ngf = {
	c:PRU ?= {
		capital = { save_scope_as = capital_state }
	}
	create_country = {
		tag = NGF
		origin = c:PRU
		state = scope:capital_state
		on_created = {
			set_up_hard_commie_gov = yes
		}
	}
	random_country = {
		limit = {
			is_player = yes
			country_has_primary_culture = cu:north_german
		}
		play_as = c:NGF
	}
	every_state = {
		limit = {
			owner = {
				country_has_primary_culture = cu:north_german
			}
		}
		set_state_owner = c:NGF
		incorporate_every_scope_state = yes
	}
	c:NGF ?= {
		create_building_and_government_buildings = yes
	}
}

create_contrabolshevist_italy = {
	c:PAP ?= {
		capital = { save_scope_as = capital_state }
	}
	create_country = {
		tag = ITA
		origin = c:SIC
		state = scope:capital_state
		on_created = {
			set_up_fascist_gov = yes
			annex_with_incorporation = c:SAR
		}
	}
	every_state = {
		limit = {
			NOT = { owner = c:ITA }
			is_homeland_of_country_cultures = c:ITA
		}
		if = {
			limit = {
				owner = { is_player = yes }
			}
			owner = { play_as = c:ITA }
		}
		set_state_owner = c:ITA
		incorporate_every_scope_state = yes
	}
	c:ITA ?= {
		ITA_afterburn = yes
	}
}
