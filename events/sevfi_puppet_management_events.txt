namespace = puppet_management

puppet_management.1 = {
	type = country_event
	hidden = yes

	# Fires on the constituent puppet itself, not globally.
	trigger = {
		is_subject = yes
		has_diplomatic_pact = {
			who = overlord
			type = constituent_republic
		}
		has_law_or_variant = law_type:law_council_republic
		overlord ?= {
			has_radical_leftist_government = no
		}
	}

	immediate = {
		overlord ?= { save_scope_as = former_overlord }
		# Referenced by sevfi_constituent_broke_away_notif_tooltip.
		save_scope_as = breakaway_puppet
		change_relations = {
			country = overlord
			value = -200
		}
		make_independent = yes
		scope:former_overlord ?= {
			post_notification = sevfi_constituent_broke_away_notif
		}
	}
}

puppet_management.2 = {
	type = country_event
	placement = root

	title = puppet_management.2.tt
	desc = puppet_management.2.dd
	flavor = puppet_management.2.ff

	cooldown = { days = normal_modifier_time }

	icon = gfx/interface/icons/invention_icons/socialism.dds

	event_image = { texture = gfx/event_pictures/puppet_management_2.dds }

	trigger = {
		has_radical_leftist_government = no
		is_subject = yes
		is_in_power_bloc = yes
		power_bloc ?= {
			has_identity = identity:identity_ideological_union
			has_principle = principle:principle_ideological_truth_3
		}
		overlord ?=  { has_radical_leftist_government = yes }
	}

	immediate = {
		overlord.capital = { save_scope_as = overlord_capital }
		root = { save_scope_as = puppet }
	}

	option = {
		name = puppet_management.2.a
		default_option = yes

		activate_law = law_type:law_council_republic
		overlord ?= {
			regime_change = scope:puppet
		}
		# Switch the existing subject pact to constituent_republic in-place
		# rather than stacking a second pact on top of the current one.
		change_subject_type = subject_type_constituent_republic
		change_relations = {
			country = overlord
			value = 200
		}
	}

	option = {
		name = puppet_management.2.b

		change_relations = {
			country = overlord
			value = -200
		}
		if = {
			limit = {
				any_scope_state = {
					owner = {
						is_adjacent_to_country = root.overlord
					}
				}
			}
			random_scope_state = {
				limit = {
					owner = {
						is_adjacent_to_country = root.overlord
					}
				}
				state_region = {
					add_claim = root.overlord
				}
			}
		}
		else = {
			# No shared border — overlord cannot press claims geographically.
			# Instead they pay a reputational price, and the puppet gains
			# a brief boost from successfully defying the ideological pressure.
			overlord ?= {
				change_infamy = 10
			}
			add_modifier = {
				name = resisted_ideological_pressure
				months = 60
			}
		}
	}
}
