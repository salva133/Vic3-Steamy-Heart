namespace = puppet_management

puppet_management.1 = {
	type = country_event
	hidden = yes

	trigger = {
		any_country = {
			is_subject = yes
			has_diplomatic_pact = {
				who = overlord
				type = constituent_republic
			}
			has_law_or_variant = law_type:law_council_republic
			overlord ?= {
				NOT = {
					has_law_or_variant = law_type:law_council_republic
				}
			}
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_subject = yes
				has_diplomatic_pact = {
					who = overlord
					type = constituent_republic
				}
				has_law_or_variant = law_type:law_council_republic
				overlord ?= {
					NOT = {
						has_law_or_variant = law_type:law_council_republic
					}
				}
			}
			change_relations = {
				country = overlord
				value = -200
			}
			make_independent = yes
		}
	}
}

puppet_management.2 = {
	type = country_event
	placement = root

	title = puppet_management.2.tt
	desc = puppet_management.2.dd
	flavor = puppet_management.2.ff

	cooldown = { days = normal_modifier_time }

	icon = gfx/interface/icons/invention_icons/socialism.dds

	event_image = {
		texture = gfx/event_pictures/puppet_management_2.dds
	}

	trigger = {
		any_country = {
			has_radical_leftist_government = no
			is_subject = yes
			is_in_power_bloc = yes
			power_bloc ?= {
				has_identity = identity:identity_ideological_union
				has_principle = principle:principle_ideological_truth_3
			}
			overlord ?=  { has_radical_leftist_government = yes }
		}
	}

	immediate = {
		overlord.capital = {
			save_scope_as = overlord_capital
		}
	}
	
	option = {
		name = puppet_management.2.a
		default_option = yes

		overlord ?= { 
			regime_change = root
		}
		change_relations = {
			country = overlord
			value = 200
		}
	}

	option = {
		name = puppet_management.2.b

		change_relations = {
			country = overlord
			value = -200
		}
	}
}
